017    package com.google.common.collect;
018    
019    import com.google.common.annotations.GwtCompatible;
020    
021    import java.util.Set;
022    
023    import javax.annotation.Nullable;
024    
025    /**
026     * A set which forwards all its method calls to another set. Subclasses should
027     * override one or more methods to modify the behavior of the backing set as
028     * desired per the <a
029     * href="http://en.wikipedia.org/wiki/Decorator_pattern">decorator pattern</a>.
030     *
031     * @author Kevin Bourrillion
032     * @since 2 (imported from Google Collections Library)
033     */
034    @GwtCompatible
035    public abstract class ForwardingSet<E> extends ForwardingCollection<E>
036        implements Set<E> {
037    
038      /** Constructor for use by subclasses. */
039      protected ForwardingSet() {}
040    
041      @Override protected abstract Set<E> delegate();
042    
043      @Override public boolean equals(@Nullable Object object) {
044        return object == this || delegate().equals(object);
045      }
046    
047      @Override public int hashCode() {
048        return delegate().hashCode();
049      }
050    }
